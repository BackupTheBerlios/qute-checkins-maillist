+function fake_diff_array(a, b) {
+    var newsel = new Array();

+function merge_arrays(a, b, b_is_select) {
+    var pos_a = 0;
+    var pos_b = 0;
+    var ret = new Array();
+    var bitem, aitem;

+

+            ret[ret.length] = aitem;

+        } else {

+            } else {

+                pos_a++;

+            }
+        }
